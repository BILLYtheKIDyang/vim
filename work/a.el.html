(defun trim (s)
  (let ((s* (replace-regexp-in-string "[ \t]*$" "" s)))
    (replace-regexp-in-string "^[ \t]*" "" s*)))

(trim " I am old ")
" I am old "

(defun take (n l)
  (cond
    ((< n 0) ())
    ((= n 0) ())
    ((null l) ())
    (t (cons (car l) (take (- n 1) (cdr l))))))

(take 3 '(1 2 3 4 5 6))  
(take 33 '(1 2 3))
